import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Group data by category_list and sum funding_total_usd
category_funding = df.groupby('category_list')['funding_total_usd'].sum().reset_index()

# Sort categories by total funding in descending order
category_funding = category_funding.sort_values(by=['funding_total_usd'], ascending=False)

# Calculate cumulative percentage
category_funding['cumulative_percentage'] = category_funding['funding_total_usd'].cumsum() / category_funding['funding_total_usd'].sum() * 100

# Create the Pareto chart with adjustments for x-axis labels
fig, ax1 = plt.subplots(figsize=(16, 8))  # Increased figure width

# Bar plot for funding_total_usd
ax1.bar(category_funding['category_list'], category_funding['funding_total_usd'], color='steelblue')
ax1.set_xlabel('Funding Categories')  # Set a generalized label
ax1.set_ylabel('Total Funding (USD)', color='steelblue')
ax1.tick_params(axis='y', labelcolor='steelblue')
ax1.set_xticklabels([])  # Remove individual category labels

# Line plot for cumulative percentage (on a secondary y-axis)
ax2 = ax1.twinx()  # Create a secondary y-axis
ax2.plot(category_funding['category_list'], category_funding['cumulative_percentage'], color='red', marker='o')
ax2.set_ylabel('Cumulative Percentage (%)', color='red')
ax2.tick_params(axis='y', labelcolor='red')

plt.title('Pareto Chart of Funding by Category')
plt.tight_layout()
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler

# Select top regions/categories (you can modify this based on your criteria)
top_categories = df.groupby('category_list')['funding_total_usd'].sum().nlargest(5).index.tolist()  # Top 5 by funding

# Filter data for selected categories
filtered_df = df[df['category_list'].isin(top_categories)]

# Group by category and get average funding rounds
category_rounds = filtered_df.groupby('category_list')['funding_rounds'].mean().reset_index()

# Normalize data for radar plot
scaler = MinMaxScaler()
normalized_data = scaler.fit_transform(category_rounds[['funding_rounds']])

# Create angles for radar plot
angles = np.linspace(0, 2 * np.pi, len(top_categories), endpoint=False)
angles = np.concatenate((angles, [angles[0]]))  # Close the loop

# Create radar plot
fig, ax = plt.subplots(figsize=(8, 8), subplot_kw={'polar': True})

# Plot data and fill area (corrected)
ax.plot(angles[:-1], normalized_data.flatten(), 'o-', linewidth=2)  # Removed extra angle and data point
ax.fill(angles[:-1], normalized_data.flatten(), alpha=0.25)  # Removed extra angle and data point

# Set thetagrids with corrected angles (corrected)
ax.set_thetagrids(angles[:-1] * 180 / np.pi, top_categories)  # Use angles[:-1]

ax.set_title('Radar Plot of Funding Rounds for Top Categories')
plt.show()

import matplotlib.pyplot as plt
import pandas as pd
import numpy as np
import matplotlib.patches as patches

# Calculate the count of startups in each funding round
funding_round_counts = df['funding_rounds'].value_counts().sort_index()

# Define the stages and their corresponding widths (adjust as needed)
stages = funding_round_counts.index.tolist()
widths = funding_round_counts.values / funding_round_counts.values.max()  # Normalize widths

# Create the funnel plot
fig, ax = plt.subplots(figsize=(10, 6))  # Adjust figure size as needed

for i, stage in enumerate(stages):
    width = widths[i]
    height = 1  # Adjust height as needed
    left = (1 - width) / 2  # Calculate left position for centering

    # Create a rectangle patch for the stage
    rect = patches.Rectangle((left, i), width, height, linewidth=1, edgecolor='black', facecolor='skyblue')

    # Add the patch to the plot
    ax.add_patch(rect)

    # Add text label for the stage
    ax.text(0.5, i + 0.5, stage, ha='center', va='center', fontsize=10)  # Adjust text properties as needed

# Set plot limits and remove ticks
ax.set_xlim(0, 1)
ax.set_ylim(0, len(stages))
ax.set_xticks([])
ax.set_yticks([])

plt.title('Funnel Plot of Startups in Each Funding Round')
plt.show()

Frequent Itemsets using FP-Growth on Startup Dataset

import pandas as pd
from mlxtend.frequent_patterns import fpgrowth
from mlxtend.preprocessing import TransactionEncoder

# Load your dataset
df = pd.read_csv('big_startup_secsees_dataset.csv')

# 1. Preprocessing:
# Handle missing values and convert categorical features to strings
df = df.fillna('')  # Fill missing values with empty strings
categorical_features = ['category_list', 'status', 'country_code']
for feature in categorical_features:
    df[feature] = df[feature].astype(str)

# 2. Create Transactions:
# Split 'category_list' into individual items and combine with other features
transactions = []
for index, row in df.iterrows():
    transaction = row['category_list'].split('|') + [row['status'], row['country_code']]
    transactions.append(transaction)

# 3. Encode Transactions:
te = TransactionEncoder()
te_ary = te.fit(transactions).transform(transactions)
encoded_df = pd.DataFrame(te_ary, columns=te.columns_)

# 4. Apply FP-Growth:
min_support = 0.05  # Adjust minimum support as needed
frequent_itemsets = fpgrowth(encoded_df, min_support=min_support, use_colnames=True)

# 5. Print Frequent Itemsets:
print("Frequent Itemsets:")
print(frequent_itemsets)
